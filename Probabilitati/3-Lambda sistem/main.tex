\begin{definition}
O familie de mulțimi $\mathcal{D}$ se numește \emph{lambda-sistem} dacă:
\begin{enumerate}
    \item $\Omega \in \mathcal{D}$.
    \item Dacă $A, B \in \mathcal{D}$ și $A \subseteq B$, atunci $B \setminus A \in \mathcal{D}$.
    \item Dacă $(A_n)_{n \ge 1}$ sunt mulțimi disjuncte două câte două din $\mathcal{D}$, atunci
    

\[
        \bigcup_{n=1}^{\infty} A_n \in \mathcal{D}.
    \]


\end{enumerate}
\end{definition}
