\begin{theorem}
\textbf{Teorema lui Liouville.}  
Fie $f : \mathbb{C} \to \mathbb{C}$ o funcție olomorfă și mărginită, adică există o constantă $M > 0$ astfel încât


\[
|f(z)| \le M \quad \text{pentru toate } z \in \mathbb{C}.
\]


Atunci funcția $f$ este constantă.
\end{theorem}
